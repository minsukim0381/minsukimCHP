Python Quick Start

import robot_api

# Connect to the robot
rb = robot_api.connect("192.168.0.10")

# Initialize motion profile
rb.set_speed(50)
rb.set_acceleration(20)

# Move to target position (x, y, z, rx, ry, rz)
rb.move_to([100, 200, 50, 0, 180, 0])

# Close connection
rb.disconnect()
